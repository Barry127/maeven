## Examples

### Basic

A basic usage of Form.

```js live withRender
// import { user, lock } from 'icon-packs/feather';
// import { code } from 'icon-packs/devicons';

const options = [
  { value: 'JavaScript' },
  { value: 'Java' },
  { value: 'PHP' },
  { value: 'Perl' },
  { value: 'C' },
  { value: 'Lua' },
  { value: 'C++' },
  { value: 'Rust' }
];

const radioOptions = [
  { label: 'Javascript', value: 'js' },
  { label: 'Typescript', value: 'ts' },
  { label: 'Coffescript', value: 'coffee' }
];

const FullForm = () => {
  const [tc, setTc] = useState(false);
  const [news, setNews] = useState(true);

  function toggleAll() {
    if (tc && news) {
      setTc(false);
      setNews(false);
    } else {
      setTc(true);
      setNews(true);
    }
  }

  return (
    <Form onSubmit={ev => ev.preventDefault()}>
      <FormFieldTextInput label="Username" icon={user} />
      <FormFieldPasswordInput label="Password" icon={lock} />
      <FormFieldSelect label="Language" options={options} />
      <FormFieldRadioGroup
        label="Flavour"
        options={radioOptions}
        name="flavour"
      />
      <FormFieldTextArea label="Comment" />
      <FormField>
        <Checkbox
          checked={tc && news}
          indeterminate={tc !== news}
          onChange={toggleAll}
        >
          Accept All
        </Checkbox>
        <Checkbox checked={tc} onChange={ev => setTc(ev.target.checked)}>
          Accept Terms & Conditions
        </Checkbox>
        <Checkbox checked={news} onChange={ev => setNews(ev.target.checked)}>
          Receive Newsletter
        </Checkbox>
      </FormField>
      <FormField>
        <Button type="submit" buttonType="primary">
          Save
        </Button>
        <Button type="reset">Cancel</Button>
      </FormField>
    </Form>
  );
};

render(<FullForm />);
```
