### Controlled

A controlled Select.

```js live withRender
const options = [
  { value: 'JavaScript' },
  { value: 'Java' },
  { value: 'PHP' },
  { value: 'Perl' },
  { value: 'C' },
  { value: 'Lua' },
  { value: 'C++' },
  { value: 'Rust' }
];

const Controlled = props => {
  const [value, setValue] = useState({ value: 'Rust' });
  return (
    <Select
      {...props}
      initialValue={value}
      onChange={ev => ev.selectedItem && setValue(ev.selectedItem)}
    >
      Value: <i>{value.value}</i>
    </Select>
  );
};

render(
  <Controlled searchable options={options} placeholder="Select something..." />
);
```

### Custom renderer

Select items can be custom rendered.

```js live withRender
// import { code, java, javascript1, perl, php, rust } from 'icon-packs/devicons';

const options = [
  { value: 'JavaScript', icon: javascript1 },
  { value: 'Java', icon: java },
  { value: 'PHP', icon: php },
  { value: 'Perl', icon: perl },
  { value: 'C', icon: code },
  { value: 'Lua', icon: code },
  { value: 'C++', icon: code },
  { value: 'Rust', icon: rust }
];

const renderItem = (item, { selected, highlighted, inputValue }) => (
  <>
    <Icon icon={item.icon} /> {item.value}
  </>
);

render(
  <Select
    renderItem={renderItem}
    searchable
    options={options}
    placeholder="Choose your skill"
  />
);
```
